121 /// Point correspondence used for calibration.
122 /// 'worldRefPos3d' is projected into camera image at 'projectedPos2d'.
123 /// The error between 'projectedPos2d' and measured marker point 'measuredPos2d' should be minimal.
124 struct CalibPoint
125 {
126  double worldRefPos3d[3]; ///< Reference 3d point (x,y,z) with respect to world coordinates.
127  double measuredPos2d[2]; ///< Measured 2d image point with respect to image coordinates.
128  double projectedPos2d[2]; ///< Image coordinates of 'worldRefPos3d' projected into image using calibrated model parameters.
129 };
