19 /// Helper class that automatically decrements reference count of a d4lib object.
20 template <class T>
21 class D4Ptr
22 {
23 public:
24  /// Constructor.
25  D4Ptr() : m_ptr(NULL) {}
26 
27  /// Copy constructor.
28  /// @param[in,out] ptr Reference to other smart pointer.
29  D4Ptr(const D4Ptr& ptr)
30  {
31  m_ptr = ptr.m_ptr;
32  d4mem_incRefCount(m_ptr);
33  }
34 
35  /// Constructor.
36  /// @param[in] ptr Valid pointer of a d4lib object.
37  /// @param[in] checkPtr Check pointer? If true and 'ptr' is NULL, an exception will be thrown.
38  /// @throw david::Exception if 'ptr'==NULL and 'checkPtr'==true.
39  explicit D4Ptr(T ptr, bool checkPtr=true) : m_ptr(ptr)
40  {
41  if (checkPtr && ptr == NULL) ThrowException(Error_InvalidArgument, "Pointer is NULL.", "D4Ptr");
42  }
43 
44  /// Destructor.
45  ~D4Ptr()
46  {
47  Reset();
48  }
49 
50  /// Convert to native handle type.
51  operator T() {return m_ptr;}
52 
53  /// Assign pointer and increment reference count.
54  /// @param[in,out] ptr Reference to other smart pointer.
55  /// @return Reference to *this.
56  D4Ptr& operator=(D4Ptr& ptr)
57  {
58  if (m_ptr == ptr.m_ptr) return *this;
59  Reset();
60  m_ptr = ptr.m_ptr;
61  if (m_ptr) d4mem_incRefCount(m_ptr);
62  return *this;
63  }
64 
65  /// Reset the smart pointer.
66  /// Decrement reference count and set internal pointer to NULL.
67  void Reset()
68  {
69  if (m_ptr) d4mem_decRefCount(m_ptr);
70  m_ptr = NULL;
71  }
72 
73 private:
74  T m_ptr; ///< Native pointer.
75 };
